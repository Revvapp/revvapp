// Revv — Firestore + Storage security rules (paste into Firebase Console).
// Deploy: firebase deploy --only firestore:rules,storage

rules_version = '2';

service cloud.firestore {
  match /databases/{database}/documents {

    function signedIn() {
      return request.auth != null;
    }

    function isOwner(uid) {
      return signedIn() && request.auth.uid == uid;
    }

    match /users/{uid} {
      allow read: if isOwner(uid);
      allow create: if isOwner(uid)
        && request.resource.data.uid == uid
        && request.resource.data.email == request.auth.token.email;
      allow update: if isOwner(uid);
      // Owner may delete their own account (App Store account-deletion requirement).
      allow delete: if isOwner(uid);
    }

    match /detailers/{uid} {
      allow read: if true;
      allow create, update: if isOwner(uid)
        && request.resource.data.uid == uid;
      allow delete: if isOwner(uid);
    }

    match /clients/{uid} {
      allow read: if isOwner(uid);
      allow create, update: if isOwner(uid)
        && request.resource.data.uid == uid;
      allow delete: if isOwner(uid);

      match /vehicles/{vehicleId} {
        allow read: if isOwner(uid);
        allow create, update: if isOwner(uid)
          && request.resource.data.ownerId == uid;
        allow delete: if isOwner(uid);
      }
    }

    match /bookings/{bookingId} {
      allow read: if signedIn()
        && (resource.data.clientId == request.auth.uid
            || resource.data.detailerId == request.auth.uid);

      // Create: caller must be the booking's client. Prevents spamming a
      // detailer's calendar with bookings someone else didn't actually request.
      allow create: if signedIn()
        && request.resource.data.clientId == request.auth.uid;

      // Update: either party on the booking, but they can't rewrite who's
      // involved or the agreed price. Status changes are role-gated: the
      // client may only cancel or sign the VIR; the detailer drives every
      // other lifecycle transition (accept/decline → VIR capture → timer
      // → complete). Stops a client from self-marking 'completed' to skip
      // the dispute window, or a detailer from rewriting 'cancelled'.
      allow update: if signedIn()
        && (resource.data.clientId == request.auth.uid
            || resource.data.detailerId == request.auth.uid)
        && request.resource.data.clientId == resource.data.clientId
        && request.resource.data.detailerId == resource.data.detailerId
        && request.resource.data.price == resource.data.price
        && (
          request.resource.data.status == resource.data.status
          || (
            resource.data.clientId == request.auth.uid
            && request.resource.data.status in ['cancelled', 'vir_signed']
          )
          || (
            resource.data.detailerId == request.auth.uid
            && request.resource.data.status in [
              'accepted', 'declined', 'vir_pending', 'vir_submitted',
              'in_progress', 'paused', 'completed'
            ]
          )
        );

      allow delete: if false;
    }

    match /invoices/{invoiceId} {
      allow read: if signedIn() && (
        resource == null ||
        resource.data.detailerId == request.auth.uid ||
        resource.data.clientId == request.auth.uid
      );
      allow create: if signedIn()
        && request.resource.data.detailerId == request.auth.uid;
      // Detailer may freely update their own invoices. Client may only flip the
      // status (and nothing else) to 'disputed' (raising a dispute) or 'released'
      // (auto-release after the dispute window closes).
      allow update: if signedIn() && (
        resource.data.detailerId == request.auth.uid
        || (
          resource.data.clientId == request.auth.uid
          && request.resource.data.diff(resource.data).affectedKeys().hasOnly(['status'])
          && request.resource.data.status in ['disputed', 'released']
        )
      );
    }

    match /reviews/{reviewId} {
      // Reviews are public so detailer profiles can display them.
      allow read: if signedIn();
      // Only the booking's own client may post, and only for a completed booking.
      allow create: if signedIn()
        && request.resource.data.clientId == request.auth.uid
        && request.resource.data.bookingId is string
        && get(/databases/$(database)/documents/bookings/$(request.resource.data.bookingId)).data.clientId == request.auth.uid
        && get(/databases/$(database)/documents/bookings/$(request.resource.data.bookingId)).data.status == 'completed';
      // Reviews are permanent.
      allow update, delete: if false;
    }

    match /disputes/{disputeId} {
      allow read: if signedIn()
        && (resource.data.clientId == request.auth.uid
            || resource.data.detailerId == request.auth.uid);
      allow create: if signedIn()
        && request.resource.data.clientId == request.auth.uid;
      // Either party may post a response / mark resolved.
      allow update: if signedIn()
        && (resource.data.clientId == request.auth.uid
            || resource.data.detailerId == request.auth.uid);
      allow delete: if false;
    }

    match /emailLookup/{emailId} {
      // Readable pre-auth so signup can resolve an existing account's type.
      allow read: if true;
      allow write: if signedIn();
    }

    match /pushTokens/{uid} {
      // Tokens are not sensitive; any signed-in user may read one to send a
      // notification, but only the owner may write their own.
      allow read: if signedIn();
      allow write: if isOwner(uid);
    }

    match /conversations/{bookingId} {
      allow read, write: if request.auth != null;

      match /messages/{messageId} {
        allow read, write: if request.auth != null;
      }
    }

  }
}

service firebase.storage {
  match /b/{bucket}/o {

    match /detailers/{uid}/{allPaths=**} {
      allow read: if true;
      allow write: if request.auth != null && request.auth.uid == uid;
    }

    match /vir-photos/{bookingId}/{allPaths=**} {
      allow read: if request.auth != null;
      allow write: if request.auth != null;
    }

    match /after-photos/{bookingId}/{allPaths=**} {
      allow read: if request.auth != null;
      allow write: if request.auth != null;
    }

    match /disputes/{uid}/{allPaths=**} {
      allow read: if request.auth != null;
      allow write: if request.auth != null && request.auth.uid == uid;
    }

  }
}
