// Revv — Firestore + Storage security rules (paste into Firebase Console).
// Deploy: firebase deploy --only firestore:rules,storage

rules_version = '2';

service cloud.firestore {
  match /databases/{database}/documents {

    function signedIn() {
      return request.auth != null;
    }

    function isOwner(uid) {
      return signedIn() && request.auth.uid == uid;
    }

    match /users/{uid} {
      allow read: if isOwner(uid);
      allow create: if isOwner(uid)
        && request.resource.data.uid == uid
        && request.resource.data.email == request.auth.token.email;
      allow update: if isOwner(uid);
      // Owner may delete their own account (App Store account-deletion requirement).
      allow delete: if isOwner(uid);
    }

    match /detailers/{uid} {
      allow read: if true;
      allow create, update: if isOwner(uid)
        && request.resource.data.uid == uid;
      allow delete: if isOwner(uid);
    }

    match /clients/{uid} {
      allow read: if isOwner(uid);
      allow create, update: if isOwner(uid)
        && request.resource.data.uid == uid;
      allow delete: if isOwner(uid);

      match /vehicles/{vehicleId} {
        allow read: if isOwner(uid);
        allow create, update: if isOwner(uid)
          && request.resource.data.ownerId == uid;
        allow delete: if isOwner(uid);
      }
    }

    match /bookings/{bookingId} {
      allow read: if signedIn()
        && (resource.data.clientId == request.auth.uid
            || resource.data.detailerId == request.auth.uid);

      // Create: caller must be the booking's client. Prevents spamming a
      // detailer's calendar with bookings someone else didn't actually request.
      allow create: if signedIn()
        && request.resource.data.clientId == request.auth.uid;

      // Update: either party on the booking, but they can't rewrite who's
      // involved or the agreed price. Status changes are role-gated: the
      // client may only cancel or sign the VIR; the detailer drives every
      // other lifecycle transition (accept/decline → VIR capture → timer
      // → complete). Stops a client from self-marking 'completed' to skip
      // the dispute window, or a detailer from rewriting 'cancelled'.
      allow update: if signedIn()
        && (resource.data.clientId == request.auth.uid
            || resource.data.detailerId == request.auth.uid)
        && request.resource.data.clientId == resource.data.clientId
        && request.resource.data.detailerId == resource.data.detailerId
        && request.resource.data.price == resource.data.price
        && (
          request.resource.data.status == resource.data.status
          || (
            resource.data.clientId == request.auth.uid
            && request.resource.data.status in ['cancelled', 'vir_signed']
          )
          || (
            resource.data.detailerId == request.auth.uid
            && request.resource.data.status in [
              'accepted', 'declined', 'vir_pending', 'vir_submitted',
              'in_progress', 'paused', 'completed'
            ]
          )
        );

      allow delete: if false;
    }

    match /invoices/{invoiceId} {
      allow read: if signedIn() && (
        resource == null ||
        resource.data.detailerId == request.auth.uid ||
        resource.data.clientId == request.auth.uid
      );
      allow create: if signedIn()
        && request.resource.data.detailerId == request.auth.uid;
      // Detailer may freely update their own invoices. Client may only flip the
      // status (and nothing else) to 'disputed' (raising a dispute) or 'released'
      // (auto-release after the dispute window closes).
      allow update: if signedIn() && (
        resource.data.detailerId == request.auth.uid
        || (
          resource.data.clientId == request.auth.uid
          && request.resource.data.diff(resource.data).affectedKeys().hasOnly(['status'])
          && request.resource.data.status in ['disputed', 'released']
        )
      );
    }

    match /reviews/{reviewId} {
      // Reviews are public so detailer profiles can display them.
      allow read: if signedIn();
      // Only the booking's own client may post, and only for a completed booking.
      // The review's doc id MUST equal its bookingId — combined with the
      // update/delete denial below, that caps it at exactly one review per
      // booking. Without this, a client could addDoc() many random-id reviews
      // for the same booking and inflate or tank a detailer's aggregate rating.
      allow create: if signedIn()
        && reviewId == request.resource.data.bookingId
        && request.resource.data.clientId == request.auth.uid
        && request.resource.data.bookingId is string
        && get(/databases/$(database)/documents/bookings/$(request.resource.data.bookingId)).data.clientId == request.auth.uid
        && get(/databases/$(database)/documents/bookings/$(request.resource.data.bookingId)).data.status == 'completed';
      // Reviews are permanent.
      allow update, delete: if false;
    }

    match /disputes/{disputeId} {
      allow read: if signedIn()
        && (resource.data.clientId == request.auth.uid
            || resource.data.detailerId == request.auth.uid);
      allow create: if signedIn()
        && request.resource.data.clientId == request.auth.uid;
      // Either party may post a response / mark resolved.
      allow update: if signedIn()
        && (resource.data.clientId == request.auth.uid
            || resource.data.detailerId == request.auth.uid);
      allow delete: if false;
    }

    match /reports/{reportId} {
      // A report is a confidential flag to the REVV trust & safety team, NOT a
      // message to the other party — so the reported user must never be able to
      // read it. Only the author can read their own reports; the team uses the
      // Admin SDK (which bypasses these rules).
      allow read: if signedIn() && resource.data.reporterId == request.auth.uid;
      // Only a party on the underlying booking may file one, stamped as
      // themselves. clientId/detailerId carry the booking parties so the team
      // knows who was reported; one of them must be the reporter.
      allow create: if signedIn()
        && request.resource.data.reporterId == request.auth.uid
        && (request.resource.data.reporterRole == 'client'
            || request.resource.data.reporterRole == 'detailer')
        && (request.resource.data.clientId == request.auth.uid
            || request.resource.data.detailerId == request.auth.uid);
      // Immutable — reports are handled by the team off-platform.
      allow update, delete: if false;
    }

    match /emailLookup/{emailId} {
      // Readable pre-auth so signup can resolve an existing account's type.
      allow read: if true;
      // A user may only write the lookup for their OWN email (the doc id is the
      // lowercased email), and only point it at their own uid. Stops a signed-in
      // user from overwriting someone else's email→userType/uid mapping and
      // hijacking how that account is resolved at sign-in.
      allow write: if signedIn()
        && request.auth.token.email != null
        && emailId == request.auth.token.email.lower()
        && request.resource.data.uid == request.auth.uid;
    }

    match /pushTokens/{uid} {
      // Notifications are sent server-side (Cloud Functions, Admin SDK), so the
      // client never needs to read another user's token. Lock read+write to the
      // owner — a client can manage its own token and nothing more. The Admin
      // SDK bypasses these rules, so the notification functions still read freely.
      allow read, write: if isOwner(uid);
    }

    // A conversation lives at conversations/{bookingId} — its doc id is the
    // booking id, so the booking is the source of truth for who may take part.
    match /conversations/{bookingId} {
      function booking() {
        return get(/databases/$(database)/documents/bookings/$(bookingId)).data;
      }
      // True only for the actual client/detailer on the underlying booking.
      // Reads the booking (not the conversation's own fields) so participation
      // can't be forged by writing the conversation doc.
      function isBookingParty() {
        return signedIn()
          && exists(/databases/$(database)/documents/bookings/$(bookingId))
          && (booking().clientId == request.auth.uid
              || booking().detailerId == request.auth.uid);
      }

      // Read/list off the conversation's own participant fields — no cross-doc
      // get(), so listing a user's threads (where clientId/detailerId == uid)
      // stays a single cheap query. Those fields are trustworthy because the
      // write rule below forces them to match the booking.
      allow read: if signedIn()
        && (resource.data.clientId == request.auth.uid
            || resource.data.detailerId == request.auth.uid);

      // Only a real booking party may write the thread doc, and they can only
      // stamp the true participant ids from the booking. Stops a stranger who
      // knows a booking id from hijacking the thread or reading its messages.
      allow create, update: if isBookingParty()
        && request.resource.data.clientId == booking().clientId
        && request.resource.data.detailerId == booking().detailerId;

      allow delete: if false;

      match /messages/{messageId} {
        // Only the two booking parties can read the thread's messages.
        allow read: if isBookingParty();
        // A party may post, but only as themselves — no spoofing senderId.
        allow create: if isBookingParty()
          && request.resource.data.senderId == request.auth.uid;
        // Messages are an immutable record.
        allow update, delete: if false;
      }
    }

  }
}

service firebase.storage {
  match /b/{bucket}/o {

    match /detailers/{uid}/{allPaths=**} {
      allow read: if true;
      allow write: if request.auth != null && request.auth.uid == uid;
    }

    match /vir-photos/{bookingId}/{allPaths=**} {
      allow read: if request.auth != null;
      allow write: if request.auth != null;
    }

    match /after-photos/{bookingId}/{allPaths=**} {
      allow read: if request.auth != null;
      allow write: if request.auth != null;
    }

    match /disputes/{uid}/{allPaths=**} {
      allow read: if request.auth != null;
      allow write: if request.auth != null && request.auth.uid == uid;
    }

  }
}
